#include<iostream>
#include<string>
#include <memory>
using namespace std;
class Float
{
	float data;

public :
	Float ()
	{
		data=0;
	}
	Float(float num)
	{
		data=num;
	}

Float operator +(float f)
{
	
	return data+f;
}
Float operator +(Float f)
{
	
	return data+f.data;
}
void show()
{
cout<<"data = "<<data<<endl;
}
};

void main()
{
	Float f1(100.2f),f2(200.4f),f3;
	f1.show();
	f2.show();
	f3.show();
	f3=f1+f2;
	f1.show();
	f2.show();
	f3.show();

}
