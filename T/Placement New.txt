#include<iostream>
using namespace std;
int main()
{
	// initial value of X
	int *X = new int(10);

	cout << "Before placement new :" << endl;
	cout << "X : " << X << endl;
//	cout << "&X : " << &X << endl;

	// Placement new changes the value of X to 100
	int *mem = new (X) int[2];
	*mem = 123;
	*(mem + 1) = 456;
	cout << "\nAfter placement new :" << endl;
	cout << "X : " << X << endl;
	cout << "mem : " << mem << endl;
	cout << "*X : " << *X << endl;
	cout << "*X : " << *(mem+1) << endl;
	cout << "*X : " << *(mem) << endl;

	return 0;
}
